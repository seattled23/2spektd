  bash install.sh

USE IN CLAUDE CODE:
  /2spektd:new "Build analytics dashboard"
  /2spektd:upgrade pkg/mycomponent go

DIRECT VALIDATION:
  bash validation/validate-component.sh <path> <language>

MCP SERVER:
  Add to ~/.claude.json:
  {
    "mcpServers": {
      "2spektd": {
        "command": "python",
        "args": ["/absolute/path/to/mcp-server/server.py"]
      }
    }
  }
